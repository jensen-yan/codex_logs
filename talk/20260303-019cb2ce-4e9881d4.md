def load_metric_specs(config_path: Optional[Path]) -> List[MetricSpec]:
    if config_path is None:
        return default_metric_specs()

    path = config_path.resolve()
    if not path.is_file():
        raise FileNotFoundError(f"metrics config not found: {path}")

    suffix = path.suffix.lower()
    if suffix == ".csv":
        with path.open(encoding="utf-8", newline="") as fp:
            rows = list(csv.DictReader(fp))
    elif suffix in {".yaml", ".yml"}:
        import yaml

        payload = yaml.safe_load(path.read_text(encoding="utf-8"))
        if isinstance(payload, dict):
            rows = payload.get("metrics", [])
        elif isinstance(payload, list):
            rows = payload
        else:
            raise ValueError("yaml config must be a list or contain a 'metrics' list")
    else:
        raise ValueError("metrics config must be .csv/.yml/.yaml")

    specs: List[MetricSpec] = []
    for idx, row in enumerate(rows):
        if not isinstance(row, dict):
            raise ValueError(f"invalid metric row type at index {idx}: {type(row)}")

        name = str(row.get("name", "")).strip()
        kind = str(row.get("kind", row.get("type", "counter"))).strip().lower() or "counter"
        numerator = str(row.get("numerator", row.get("counter", ""))).strip()
        denominator = str(row.get("denominator", "")).strip()
        scale = safe_float(row.get("scale", 1.0), default=1.0)
        description = str(row.get("description", "")).strip()

        if not name:
            raise ValueError(f"metric row {idx} missing 'name'")
        if not numerator:
            raise ValueError(f"metric row {idx} missing 'numerator' or 'counter'")
        if kind not in {"counter", "rate"}:
            raise ValueError(f"metric '{name}' has unsupported kind '{kind}'")
        if kind == "rate" and not denominator:
            raise ValueError(f"metric '{name}' is rate but denominator is empty")

        specs.append(
            MetricSpec(
                name=name,
                kind=kind,
                numerator=numerator,
                denominator=denominator,
                scale=scale,
                description=description,
            )
        )

    if not specs:
        raise ValueError("metrics config is empty")
    return specs


def find_top_mispredict_csv(case_dir: Path) -> Optional[Path]:
    candidates = [
        case_dir / "topMispredictsByBranch.csv",
        case_dir / "coremark" / "topMispredictsByBranch.csv",
    ]
    for path in candidates:
        if path.is_file():
            return path

    fallback = sorted(case_dir.rglob("topMispredictsByBranch.csv"))
    return fallback[0] if fallback else None


def parse_hotspots(csv_path: Path, top_n: int, min_total: int) -> List[HotspotEntry]:
    rows: List[HotspotEntry] = []
    with csv_path.open(encoding="utf-8", newline="") as fp:
        reader = csv.DictReader(fp)
        for row in reader:
            total = safe_int(row.get("total", 0))
            if total < min_total:
                continue

            rows.append(
                HotspotEntry(
                    pc=str(row.get("pc", "")),
                    mispredicts=safe_int(row.get("mispredicts", 0)),
                    total=total,
                    mis_permil=safe_float(row.get("misPermil", 0.0)),
                    dir_miss=safe_int(row.get("dirMiss", 0)),
                    tgt_miss=safe_int(row.get("tgtMiss", 0)),
                    no_pred_miss=safe_int(row.get("noPredMiss", 0)),
                )
            )
    rows.sort(key=lambda x: x.mispredicts, reverse=True)
    return rows[:top_n]


def render_summary_markdown(
    summary_path: Path,
    results: List[CaseMetrics],
    metric_names: List[str],
    config_name: str,
) -> None:
    ok_count = sum(1 for r in results if r.analysis_ok)
    missing_count = sum(1 for r in results if not r.stats_exists)

    lines: List[str] = []
    lines.append("# BPU Metrics Summary")
    lines.append("")
    lines.append(f"- generated_at: `{now_iso()}`")
    lines.append(f"- config: `{config_name}`")
    lines.append(f"- scanned_cases: **{len(results)}**")
    lines.append(f"- analysis_ok: **{ok_count}**")
    lines.append(f"- missing_stats: **{missing_count}**")
    lines.append("")
    lines.append("| case | ok | top_hotspot | errors |")
    lines.append("| --- | --- | --- | --- |")
    for result in sorted(results, key=lambda x: x.case_name):
        hotspot = result.top_hotspot
        hotspot_str = "-"
        if hotspot is not None:
            hotspot_str = f"{hotspot.pc}/{hotspot.mispredicts}/{hotspot.total}/{hotspot.mis_permil:.3f}"
        lines.append(
            f"| {result.case_name} | {'yes' if result.analysis_ok else 'no'} | {hotspot_str} | {len(result.errors)} |"
        )

    lines.append("")
    lines.append("## Metrics")
    for metric_name in metric_names:
        lines.append(f"- `{metric_name}`")

    summary_path.write_text("\n".join(lines) + "\n", encoding="utf-8")


class GEM5Runner:
    def __init__(

        self.common_params = ["--raw-cpt"]
        self.kmhv3_params = kmhv3_params

        self.top_n = top_n

        self.metric_specs = load_metric_specs(self.metrics_config_path)

        logging.basicConfig(level=logging.INFO, format="%(asctime)s - %(levelname)s - %(message)s")
        self.logger = logging.getLogger(__name__)

        self.slices: Dict[str, str] = {

    def load_frontend_tests(self) -> None:
        am_home = os.environ.get("AM_HOME")
        if not am_home:
            self.logger.warning("AM_HOME is not set; skip frontend test discovery")
            return

        base_dir = Path(am_home) / "tests" / "frontendtest"
        build_dirs = [
            base_dir / "build",
            base_dir / "br_target_test" / "build",
            base_dir / "cond_br_test" / "build",
            base_dir / "mgsc_test" / "build",
        ]

        discovered = 0
        for directory in build_dirs:
            if not directory.exists():
                self.logger.warning("Frontend test directory not found: %s", directory)
                continue

            for binary in directory.glob("*-riscv64-xs.bin"):
                slice_name = binary.stem
                suffix = "-riscv64-xs"
                if slice_name.endswith(suffix):
                    slice_name = slice_name[: -len(suffix)]

                if slice_name not in self.slices:
                    discovered += 1
                self.slices[slice_name] = str(binary)

        if discovered:
            self.logger.info("Discovered %d frontend tests via AM_HOME", discovered)

    def generate_sim_configs(self) -> List[SimConfig]:
        configs: List[SimConfig] = []
        for slice_name, checkpoint in self.slices.items():
            if not self.skip_ref:
                configs.append(
                    SimConfig(
                        binary="gem5.opt.ref",
                        slice_name=slice_name,
                        checkpoint=checkpoint,
                        outdir=self.debug_dir / f"{slice_name}_ref",
                        args=["--enable-bp-db=tage"],
                    )
                )
            configs.append(
                SimConfig(
                    binary="gem5.opt",
                    slice_name=slice_name,
                    checkpoint=checkpoint,
                    outdir=self.debug_dir / f"{slice_name}_opt",
                    args=["--enable-bp-db=tage"],
                )
            )
        return configs

    def run_simulation(self, config: SimConfig) -> bool:
        config.outdir.mkdir(parents=True, exist_ok=True)
        stdout_file = config.outdir / "gem5.stdout"
        stderr_file = config.outdir / "gem5.stderr"

        cmd: List[str] = [
            str(GEM5_BUILD_DIR / config.binary),
            "--outdir",
            str(config.outdir),
            str(KMHV3_CONFIG),
            "--generic-rv-cpt",
            str(config.checkpoint),
            *self.common_params,
            *config.args,
        ]
        for param in self.kmhv3_params:
            cmd.extend(["-P", param])

        self.logger.info("Run %s with %s", config.slice_name, config.binary)
        with stdout_file.open("w", encoding="utf-8") as out, stderr_file.open("w", encoding="utf-8") as err:
            proc = subprocess.run(cmd, stdout=out, stderr=err, text=True)

        if proc.returncode == 0:
            return True

        error_text = stderr_file.read_text(encoding="utf-8", errors="ignore").strip()
        self.logger.error("Simulation failed: %s %s: %s", config.slice_name, config.binary, error_text)
        return False

    def run_all(self) -> None:
        configs = self.generate_sim_configs()
        success = 0
        failed = 0

        with concurrent.futures.ThreadPoolExecutor(max_workers=self.max_workers) as executor:
            future_map = {executor.submit(self.run_simulation, config): config for config in configs}
            for future in concurrent.futures.as_completed(future_map):
                try:
                    if future.result():
                        success += 1
                    else:
                        failed += 1
                except Exception as exc:
                    failed += 1
                    config = future_map[future]
                    self.logger.error("Simulation exception on %s: %s", config.slice_name, exc)

        self.logger.info("Simulation done. success=%d failed=%d", success, failed)

    def discover_analysis_cases(self) -> List[Path]:
        if not self.debug_dir.exists():
            self.logger.warning("Debug directory does not exist: %s", self.debug_dir)
            return []

        case_dirs = [p for p in self.debug_dir.iterdir() if p.is_dir()]
        if self.analysis_target == "opt":
            case_dirs = [p for p in case_dirs if p.name.endswith("_opt")]
        return sorted(case_dirs, key=lambda p: p.name)

    def analyze_case(self, case_dir: Path) -> CaseMetrics:
        case_name = case_dir.name
        stats_path = case_dir / "stats.txt"
        result = CaseMetrics(
            case_name=case_name,
            case_dir=case_dir,
            stats_path=stats_path,
            stats_exists=stats_path.is_file(),
        )

        if not result.stats_exists:
            return result

        try:
            stats = parse_stats_last_block(stats_path)
        except Exception as exc:
            result.errors.append(f"parse stats failed: {exc}")
            return result

        missing: List[str] = []
        metric_map: Dict[str, float] = {}
        metric_details: List[Dict[str, Any]] = []
        for spec in self.metric_specs:
            try:
                value, detail, miss = evaluate_metric(stats, spec)
            except Exception as exc:
                result.errors.append(f"metric '{spec.name}' failed: {exc}")
                continue
            metric_map[spec.name] = value
            metric_details.append(detail)
            missing.extend(miss)

        result.metrics = metric_map
        result.metric_details = metric_details
        result.missing_counters = sorted(set(missing))

        csv_path = find_top_mispredict_csv(case_dir)
        if csv_path is not None:
            try:
                result.hotspots = parse_hotspots(csv_path, self.top_n, self.min_total)
            except Exception as exc:
                result.errors.append(f"parse hotspot csv failed: {exc}")

        payload = {
            "case": case_name,
            "stats_path": str(stats_path),
            "generated_at": now_iso(),
            "metrics_config": str(self.metrics_config_path) if self.metrics_config_path else "built-in",
            "metrics": result.metrics,
            "metric_details": result.metric_details,
            "missing_counters": result.missing_counters,
            "hotspots": [
                {
                    "pc": h.pc,
                    "mispredicts": h.mispredicts,
                    "total": h.total,
                    "misPermil": h.mis_permil,
                    "dirMiss": h.dir_miss,
                    "tgtMiss": h.tgt_miss,
                    "noPredMiss": h.no_pred_miss,
                }
                for h in result.hotspots
            ],
            "errors": result.errors,
        }

        out_json = case_dir / "bpu_metrics.json"
        out_json.write_text(json.dumps(payload, indent=2, ensure_ascii=False) + "\n", encoding="utf-8")
        result.output_json_path = out_json

        # Keep legacy filename for compatibility with existing workflows.
        legacy_json = case_dir / "bpu_module_report.json"
        legacy_json.write_text(json.dumps(payload, indent=2, ensure_ascii=False) + "\n", encoding="utf-8")
        return result

    def write_batch_outputs(self, results: List[CaseMetrics]) -> Tuple[Path, Path, Optional[Path]]:
        metrics_order = [spec.name for spec in self.metric_specs]
        summary_json = self.debug_dir / "bpu_metrics_summary.json"
        summary_csv = self.debug_dir / "bpu_metrics_summary.csv"

        json_payload = {
            "generated_at": now_iso(),
            "metrics_config": str(self.metrics_config_path) if self.metrics_config_path else "built-in",
            "metric_names": metrics_order,
            "cases": [
                {
                    "case": r.case_name,
                    "stats_exists": r.stats_exists,
                    "analysis_ok": r.analysis_ok,
                    "metrics": r.metrics,
                    "missing_counters": r.missing_counters,
                    "errors": r.errors,
                    "top_hotspot": (
                        {
                            "pc": r.top_hotspot.pc,
                            "mispredicts": r.top_hotspot.mispredicts,
                            "total": r.top_hotspot.total,
                            "misPermil": r.top_hotspot.mis_permil,
                            "dirMiss": r.top_hotspot.dir_miss,
                            "tgtMiss": r.top_hotspot.tgt_miss,
                            "noPredMiss": r.top_hotspot.no_pred_miss,
                        }
                        if r.top_hotspot
                        else None
                    ),
                }
                for r in sorted(results, key=lambda x: x.case_name)
            ],
        }
        summary_json.write_text(json.dumps(json_payload, indent=2, ensure_ascii=False) + "\n", encoding="utf-8")

        csv_headers = [
            "case",
            "stats_exists",
            "analysis_ok",
            "top_hotspot_pc",
            "top_hotspot_mispredicts",
            "top_hotspot_total",
            "top_hotspot_mis_permil",
            "missing_counter_count",
            "error_count",
            *metrics_order,
        ]

        with summary_csv.open("w", encoding="utf-8", newline="") as fp:
            writer = csv.DictWriter(fp, fieldnames=csv_headers)
            writer.writeheader()
            for result in sorted(results, key=lambda x: x.case_name):
                row = {
                    "case": result.case_name,
                    "stats_exists": int(result.stats_exists),
                    "analysis_ok": int(result.analysis_ok),
                    "top_hotspot_pc": "",
                    "top_hotspot_mispredicts": 0,
                    "top_hotspot_total": 0,
                    "top_hotspot_mis_permil": 0.0,
                    "missing_counter_count": len(result.missing_counters),
                    "error_count": len(result.errors),
                }
                if result.top_hotspot is not None:
                    row.update(
                        {
                            "top_hotspot_pc": result.top_hotspot.pc,
                            "top_hotspot_mispredicts": result.top_hotspot.mispredicts,
                            "top_hotspot_total": result.top_hotspot.total,
                            "top_hotspot_mis_permil": result.top_hotspot.mis_permil,
                        }
                    )
                for metric_name in metrics_order:
                    row[metric_name] = result.metrics.get(metric_name, "")
                writer.writerow(row)

        md_path: Optional[Path] = None
        if self.emit_markdown:
            md_path = self.debug_dir / "bpu_batch_summary.md"
            config_name = str(self.metrics_config_path) if self.metrics_config_path else "built-in"
            render_summary_markdown(md_path, results, metrics_order, config_name)

        return summary_json, summary_csv, md_path

    def run_batch_analysis(self) -> List[CaseMetrics]:
        cases = self.discover_analysis_cases()
        if not cases:
            self.logger.warning("No case dirs found under %s", self.debug_dir)
            self.write_batch_outputs([])
            return []

        self.logger.info(
            "Start analysis: cases=%d workers=%d target=%s",
            len(cases),
            self.analysis_workers,
            self.analysis_target,
        )

        results: List[CaseMetrics] = []
        with concurrent.futures.ThreadPoolExecutor(max_workers=self.analysis_workers) as executor:
            future_map = {executor.submit(self.analyze_case, case): case for case in cases}
            for future in concurrent.futures.as_completed(future_map):
                case = future_map[future]
                try:
                    result = future.result()
                except Exception as exc:
                    result = CaseMetrics(
                        case_name=case.name,
                        case_dir=case,
                        stats_path=case / "stats.txt",
                        stats_exists=(case / "stats.txt").is_file(),
                    )
                    result.errors.append(f"unhandled analysis exception: {exc}")
                results.append(result)

                if not result.stats_exists:
                    self.logger.warning("Skip %s: missing stats.txt", result.case_name)
                elif result.analysis_ok:
                    self.logger.info("Analysis OK: %s", result.case_name)
                else:
                    self.logger.warning("Analysis has errors: %s", result.case_name)

        summary_json, summary_csv, md_path = self.write_batch_outputs(results)
        self.logger.info("Wrote summary json: %s", summary_json)
        self.logger.info("Wrote summary csv: %s", summary_csv)
        if md_path is not None:
            self.logger.info("Wrote summary markdown: %s", md_path)
        return results

def main() -> int:
    args = build_arg_parser().parse_args()

    runner = GEM5Runner(
        max_workers=args.max_workers,
        debug_dir=args.debug_dir,
        kmhv3_params=args.param,
        skip_ref=args.skip_ref,
        analysis_workers=args.analysis_workers,
        analysis_target=args.analysis_target,
        top_n=args.top_n,
        min_total=args.min_total,
        metrics_config=args.metrics_config,
        emit_markdown=args.emit_markdown,
    )

    if not args.analyze_only:
        if args.slices:
            runner.slices = {name: path for name, path in runner.slices.items() if name in args.slices}

                runner.logger.error("No valid slices specified")

        runner.logger.info("Analyze-only mode: skip simulation")

    if args.skip_analysis:
        runner.logger.info("Skip analysis due to --skip-analysis")
        return 0

    runner.run_batch_analysis()
    return 0

END = "---------- End Simulation Statistics   ----------"

@dataclass
class CaseRecord:
    case_path: str
    stats_path: str
    values: Dict[str, float]
    missing: List[str]

def now_iso() -> str:
    return datetime.now(timezone.utc).isoformat()


def parse_last_stats_block(path: Path) -> Dict[str, float]:
    lines = path.read_text(encoding="utf-8", errors="ignore").splitlines()
    blocks: List[List[str]] = []
    in_block = False
    current: List[str] = []

    for line in lines:
        stripped = line.strip()
        if stripped == BEGIN:
            in_block = True
            current = []
            continue
        if stripped == END and in_block:
            blocks.append(current)
            in_block = False
            continue
        if in_block:
            current.append(line)

    target = blocks[-1] if blocks else lines
    stats: Dict[str, float] = {}
    for line in target:
        if not line or line.startswith("-"):
            continue
        parts = line.split()
        if len(parts) < 2:
            continue
        key, value = parts[0], parts[1]
        try:
            stats[key] = float(value)
        except ValueError:
            continue
    return stats


def load_counters(path: Path) -> List[str]:
    suffix = path.suffix.lower()
    if suffix in {".txt", ""}:
        counters = [line.strip() for line in path.read_text(encoding="utf-8").splitlines()]
        counters = [c for c in counters if c and not c.startswith("#")]
        if not counters:
            raise ValueError(f"no counters found in {path}")
        return counters

    if suffix in {".yml", ".yaml"}:
        import yaml

        payload = yaml.safe_load(path.read_text(encoding="utf-8"))
        if isinstance(payload, list):
            counters = [str(x).strip() for x in payload if str(x).strip()]
        elif isinstance(payload, dict):
            raw = payload.get("counters", [])
            counters = [str(x).strip() for x in raw if str(x).strip()]
        else:
            raise ValueError("yaml must be list or object with 'counters'")
        if not counters:
            raise ValueError(f"no counters found in {path}")
        return counters

    if suffix == ".csv":
        counters: List[str] = []
        with path.open(encoding="utf-8", newline="") as fp:
            reader = csv.DictReader(fp)
            if reader.fieldnames is None:
                raise ValueError(f"invalid csv with no header: {path}")
            column = "counter" if "counter" in reader.fieldnames else reader.fieldnames[0]
            for row in reader:
                value = str(row.get(column, "")).strip()
                if value:
                    counters.append(value)
        if not counters:
            raise ValueError(f"no counters found in {path}")
        return counters

    raise ValueError("counter file must be .txt/.yml/.yaml/.csv")

    record = CaseRecord(
        case_path=str(case_rel),
        stats_path=str(stats_path),
        values={},
        missing=[],
        errors=[],
    )

    try:
        stats = parse_last_stats_block(stats_path)
    except Exception as exc:
        record.errors.append(f"parse stats failed: {exc}")
        return record

    values: Dict[str, float] = {}
    missing: List[str] = []
    for counter in counters:
        if counter in stats:
            values[counter] = stats[counter]
        else:
            missing.append(counter)

    record.values = values
    record.missing = missing
    return record

# BPU 计数器分析技能（极简）

## 何时使用
- 你已经跑完 gem5，只想批量提取 BPU 相关原始计数器。
- 你不想在脚本里做复杂推导，只要原始值。
- 你需要机器可读结果给后续脚本/表格处理。

## 核心原则
- 只提取原始 counter，不做公式推导。
- 配置文件只写 counter 名称。
- 目录结构不做强约束，递归扫描 `stats.txt`。

## 入口脚本
- `.codex/skills/frontend-pmu-analysis/scripts/analyze_bpu_counters.py`

## 默认配置
- `.codex/skills/frontend-pmu-analysis/configs/bpu_counters.txt`

## 用法
```bash
python3 .codex/skills/frontend-pmu-analysis/scripts/analyze_bpu_counters.py \
  --debug-dir /tmp/debug/tage-new8
```

指定自定义计数器文件：

```bash
python3 .codex/skills/frontend-pmu-analysis/scripts/analyze_bpu_counters.py \
  --debug-dir /tmp/debug/tage-new8 \
  --counters-file /path/to/my_counters.txt
```

## 输出
- `bpu_counters_summary.json`
- `bpu_counters_summary.csv`

输出字段只包含：
- case 路径
- stats 路径
- `values`（命中的计数器和值）
- `missing`（缺失计数器）
- `errors`（解析错误）

## 计数器文件格式
推荐 `txt`（每行一个）：

```txt
system.cpu.ipc
system.cpu.commit.branchMispredicts
system.cpu.commit.branches
```

也支持：
- `yaml`：`counters: [ ... ]` 或直接列表
- `csv`：第一列或 `counter` 列为计数器名
EOF'
